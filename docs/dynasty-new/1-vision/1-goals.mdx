import { readFileSync } from 'node:fs';
import { resolve, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import yaml from 'js-yaml';
import GoalsMatrix from '@charts/GoalsMatrix.astro';

export const __filename = fileURLToPath(import.meta.url);
export const __dirname  = dirname(__filename);

export const goalsData = yaml.load(
  readFileSync(resolve(__dirname, '../6-meta/goals.yaml'), 'utf8')
);

export const { pillars, decisions } = goalsData;

export const colors = [
  { dark: '#f07070', light: '#c73030' },
  { dark: '#5dd4c8', light: '#1a9e94' },
  { dark: '#f5c05a', light: '#c9921a' },
  { dark: '#a078e8', light: '#5828b0' },
  { dark: '#70d490', light: '#2a9e50' },
];

export const pillarTotals = pillars.map(p => {
  const score = decisions
    .filter(d => d.pillars.includes(p.id))
    .reduce((s, d) => s + d.impact, 0);
  return { ...p, score };
});

export const grandTotal = pillarTotals.reduce((s, p) => s + p.score, 0);

This page is the **north star** for all design decisions in Hoop Dynasty.

Every mechanic, system, and trade-off should ultimately serve one or more of the five pillars below.
When a decision is made, it is tagged against these pillars in `6-meta/goals.yaml` with an impact score.
The chart visualises where weight has been intentionally placed (**intent**) and where it has actually landed based on the decisions made so far (**actual**).

## Priority matrix

> The **arc width** of each segment reflects the pillar's *actual weight* — derived from the impact scores of all resolved design decisions.
> Intent weights are listed in the pillar descriptions below.

## Alignment summary

<table class="alignment-table">
  <thead>
    <tr><th>Pillar</th><th class="right">Decision score</th><th class="right">Actual %</th><th class="right">Intent %</th><th class="right">Gap</th></tr>
  </thead>
  <tbody>
    {pillarTotals.map(p => {
      const actualPct = (p.score / grandTotal) * 100;
      const gap       = actualPct - p.intent_weight;
      const gapStr    = (gap > 0 ? '+' : '') + gap.toFixed(1) + 'pp';
      const gapClass  = Math.abs(gap) < 1.5 ? 'gap-ok' : (gap > 0 ? 'gap-over' : 'gap-under');
      return (
        <tr key={p.id}>
          <td>{p.name}</td>
          <td class="right">{p.score}</td>
          <td class="right">{actualPct.toFixed(1)}%</td>
          <td class="right">{p.intent_weight.toFixed(1)}%</td>
          <td class={`right ${gapClass}`}>{gapStr}</td>
        </tr>
      );
    })}
  </tbody>
</table>

> **Re-deriving weights:** run `node scripts/derive-weights.js` after adding new decisions
> to `6-meta/goals.yaml`. Copy the output back into the `actual_weight` fields.
> See `6-meta/weight-derivation-prompt.md` for the full scoring methodology.
